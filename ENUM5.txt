MACH:=function(d)
  local i,j,K,L,n;
  n:=2^d;K:=[];L:=[];
  for i in [1..n-1] do
    if not(IsBound(K[i])) then
      if IsOddInt(i) then K[i]:=i+2; K[i+2]:=i; else K[i]:=i; fi;
      fi; 
    od;
  for i in [1..n-1] do
    j:=2*i;
    if j>=n then j:=j mod n +1; fi;
    L[i]:=j;
    od;
  return Group(PermList(K),PermList(L));
  end;
